class HomeTheaterFacade {
    -amp:Amplifier
    -tuner:Tuner
    -dvd:DvdPlayer
    -projector:Projector
    +HomeTheaterFacade(Amplifier,Tuner,DvdPlayer,Projector)
    +watchMovie(String):void
    +endMovie():void
    +listenToRadio(double):void
    +endRadio():void
}

    description:String
    tuner:Tuner
    dvd:DvdPlayer
    +on()
    +off()
    +setDvd(DvdPlayer)
    +setTuer(Tuner)
    +setVolume(int)
}

    description:String
    amplifier:Amplifier
    frequency:double
    +Tuner(String, Amplifier)
    +on()
    +off()
    +setFrequency(double)
}
class DvdPlayer {
    description:String
    amplifier:Amplifier
    movie:String
    on()
    off()
    play(String)
    stop()
    eject()
}

    +on()
    +off()
